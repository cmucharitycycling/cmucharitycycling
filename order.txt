  function init() {
    const list = document.getElementById('product-list');
    PRODUCTS.forEach((p, pIdx) => {
      let variantHtml = '';
      p.variants.forEach((v, vIdx) => {
        let sizeHtml = v.sizes.map(s => `
          <div class="size-item">
            <span class="size-label">${s}</span>
            <input type="number" min="0" class="qty" data-p="${pIdx}" data-v="${vIdx}" data-s="${s}" placeholder="0">
          </div>
        `).join('');
        variantHtml += `<div class="gender-label" style="${v.gender === 'หญิง' ? 'background:#f1c40f; color:#2c263f;' : ''}">${v.gender}</div><div class="size-grid">${sizeHtml}</div>`;
      });
      list.innerHTML += `
        <div class="product-card">
          <img src="${p.img}" class="product-img">
          <div class="product-info">
            <div class="product-name">${p.name}</div>
            <div class="product-price">ราคา ${p.price.toLocaleString()} บาท</div>
            ${variantHtml}
          </div>
        </div>`;
    });
    document.querySelectorAll('.qty').forEach(i => i.addEventListener('input', calc));
  }

  function calc() {
    let total = 0;
    document.querySelectorAll('.qty').forEach(i => {
      let q = parseInt(i.value) || 0;
      if (q > 0) total += q * PRODUCTS[i.dataset.p].price;
    });
    let ship = parseInt(document.querySelector('input[name="ship"]:checked').value);
    if (total > 0) total += ship;
    document.getElementById('display_total').innerText = total.toLocaleString();
    return total;
  }

  async function sendData() {
    const btn = document.getElementById('btn');
    let total = calc();

    if (total === 0) return alert("กรุณาเลือกสินค้าอย่างน้อย 1 ชิ้น");
    if (!document.getElementById('name').value || !document.getElementById('tel').value) return alert("กรุณากรอกชื่อและเบอร์โทรศัพท์");
    const shipVal = parseInt(document.querySelector('input[name="ship"]:checked').value);
    if (shipVal === 50 && !document.getElementById('address').value) return alert("กรุณากรอกที่อยู่จัดส่ง");
    if (document.getElementById('fileInput').files.length === 0) return alert("กรุณาแนบสลิปโอนเงิน");

    btn.disabled = true;
    btn.innerText = "⏳ กำลังบันทึกข้อมูล...";

    let details = [];
    document.querySelectorAll('.qty').forEach(i => {
      let q = parseInt(i.value) || 0;
      if (q > 0) {
        let p = PRODUCTS[i.dataset.p];
        let v = p.variants[i.dataset.v];
        details.push(`${p.name}(${v.gender})-${i.dataset.s}(${q})`);
      }
    });

    try {
      const file = document.getElementById('fileInput').files[0];
      const base64 = await new Promise(r => {
        const rd = new FileReader();
        rd.onload = () => r(rd.result.split(',')[1]);
        rd.readAsDataURL(file);
      });

      const payload = {
        name: document.getElementById('name').value,
        tel: document.getElementById('tel').value,
        address: document.getElementById('address').value,
        total: total,
        details: details.join(", "),
        fileData: base64,
        fileName: file.name,
        mimeType: file.type
      };

      // ✅ ใช้ mode: 'no-cors' เพื่อแก้ปัญหา CORS ใน Google Sites
      await fetch(WEB_APP_URL, {
        method: "POST",
        mode: "no-cors",
        headers: { "Content-Type": "text/plain" },
        body: JSON.stringify(payload)
      });

      // no-cors ไม่สามารถอ่าน response ได้ → แสดง success โดยตรง
      document.body.innerHTML = `
        <div style="text-align:center; padding:100px 20px; color:white; font-family:'Kanit',sans-serif;">
          <h1 style="font-size:80px; margin:0;">✅</h1>
          <h2 style="font-size:32px;">สั่งซื้อสำเร็จ!</h2>
          <p style="font-size:18px;">เราได้รับข้อมูลของคุณเรียบร้อยแล้ว </p>
          <button onclick="location.reload()" style="margin-top:20px; padding:12px 30px; background:#f1c40f; border:none; border-radius:8px; cursor:pointer; font-family:'Kanit'; font-size:18px; font-weight:600;">🔄 กลับหน้าหลัก</button>
        </div>`;

    } catch (e) {
      alert("เกิดข้อผิดพลาด: " + e.message);
      btn.disabled = false;
      btn.innerText = "🚀 ยืนยันการสั่งซื้อ";
    }
  }

  init();
